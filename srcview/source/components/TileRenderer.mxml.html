                autoDrawBackground="true"
                click="itemRenderer_clickHandler(event)">

    </s:layout>
    

    <fx:Script>
        <![CDATA[
            import org.tomasino.tracking.TrackingEvent;
            
            protected function itemRenderer_clickHandler(event:MouseEvent):void
            {
                var t:TrackingEvent = new TrackingEvent ();
                t.data.exitLink = data.url;
                t.data.googleCategory = 'Outbound Links';
                t.data.googleClickName = id;
                dispatchEvent (t);
            }
        ]]>
    </fx:Script>

    <mx:Image id="img" source="{data.image}" width="80%" height="80%" smoothBitmapContent="true"/>
