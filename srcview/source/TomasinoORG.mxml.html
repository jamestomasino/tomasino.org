               creationComplete="initApp(event)" viewSourceURL="../srcview/index.html">
    <s:layout>
        <s:VerticalLayout horizontalAlign="center"  
                          paddingTop="5" paddingBottom="5" paddingLeft="5" paddingRight="5" />
    </s:layout>
    <s:List id="tileList"
                 dataProvider="{tileItems}"
                 itemRenderer="components.TileRenderer"
                 dragEnabled="false" 
                 horizontalCenter="0"
                 verticalCenter="0"
                 height="100%"
                 width="100%" 
                 rollOverColor="#EEEEEE">
        <s:layout>
            <s:TileLayout horizontalGap="5"
                          verticalGap="10" />
        </s:layout>
    </s:List>
    
    <fx:Script>
        <![CDATA[
            import mx.collections.ArrayCollection;
            import mx.events.FlexEvent;
            import mx.rpc.events.FaultEvent;
            import mx.rpc.events.ResultEvent;
            
            import org.tomasino.tracking.TrackingEvent;
            import org.tomasino.tracking.TrackingManager;
            import org.tomasino.tracking.types.GoogleType;
            
            import vo.Item;
            
            [Bindable]public var tileItems:ArrayCollection = new ArrayCollection();
            private var _tracker:TrackingManager;
            
            protected function initApp(event:FlexEvent):void
            {
                // Test for display list
                if (this.stage)
                {
                    onAdd(null);
                }
                else
                {
                    addEventListener( Event.ADDED_TO_STAGE, onAdd );
                }
            }
            
            protected function onAdd ( event:Event ):void
            {
                // Setup Tracker
                _tracker = TrackingManager.instance;
                var googleType:GoogleType = new GoogleType ();
                _tracker.init ( googleType );
                this.stage.addEventListener ( TrackingEvent.TRACK, _tracker.track );
                
                // Load Data
                linksHTTP.send();
            }
            protected function linksHTTP_faultHandler(event:FaultEvent):void
            {
                // TODO Auto-generated method stub
                trace ('poop');
            }

            protected function linksHTTP_resultHandler(event:ResultEvent):void
            {
                // TODO Auto-generated method stub
                var items:XMLList = event.result..item;
                for (var i:int = 0; i < items.length(); ++i)
                {
                    var item:XML = items[i];
                    
                    var tileItem:Item = new Item();
                    tileItem.id = item.id.toString();
                    tileItem.image = 'http://www.tomasino.org/' + item.image.toString();
                    tileItem.url = item.url.toString();
                    tileItem.title = item.title.toString();
                    
                    tileItems.addItem (tileItem);
                }
                
            }

        ]]>
    </fx:Script>
    
    <fx:Declarations>
        <!-- Place non-visual elements (e.g., services, value objects) here -->
        <s:HTTPService id="linksHTTP" url="http://www.tomasino.org/links.xml" resultFormat="e4x" fault="linksHTTP_faultHandler(event)" result="linksHTTP_resultHandler(event)" />
    </fx:Declarations>
    
</s:Application>
